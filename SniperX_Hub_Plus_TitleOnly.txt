local ScreenGui = Instance.new("ScreenGui")
local Main = Instance.new("Frame")
local UICorner = Instance.new("UICorner")
local UIGrid = Instance.new("UIGridLayout")
local title = Instance.new("TextLabel")

-- BotÃµes
local buttons = {
    {Text="Kill All", Func=function()
        for _,v in pairs(game.Players:GetPlayers()) do
            if v ~= game.Players.LocalPlayer and v.Character and v.Character:FindFirstChild("Humanoid") then
                v.Character.Humanoid.Health = 0
            end
        end
    end},
    {Text="ESP Players", Func=function()
        for _,v in pairs(game.Players:GetPlayers()) do
            if v ~= game.Players.LocalPlayer and v.Character then
                local box = Instance.new("BillboardGui", v.Character)
                box.Size = UDim2.new(0,100,0,40)
                box.AlwaysOnTop = true
                local tag = Instance.new("TextLabel", box)
                tag.Text = v.Name
                tag.Size = UDim2.new(1,0,1,0)
                tag.BackgroundTransparency = 1
                tag.TextColor3 = Color3.new(1,0,0)
                tag.TextScaled = true
            end
        end
    end},
    {Text="TP All to Me", Func=function()
        local root = game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        for _,v in pairs(game.Players:GetPlayers()) do
            if v ~= game.Players.LocalPlayer and v.Character then
                v.Character:SetPrimaryPartCFrame(root.CFrame + Vector3.new(math.random(-5,5), 0, math.random(-5,5)))
            end
        end
    end},
    {Text="Speed x3", Func=function()
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = 48
    end},
    {Text="Fly", Func=function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/BladeTheWarrior/FlyGui/main/FlyGui.lua"))()
    end},
    {Text="Destroy GUI", Func=function()
        ScreenGui:Destroy()
    end},
}

-- Criar botÃµes
for _,btn in ipairs(buttons) do
    local b = Instance.new("TextButton")
    b.Text = btn.Text
    b.Font = Enum.Font.GothamBold
    b.TextScaled = true
    b.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    b.TextColor3 = Color3.fromRGB(255, 255, 255)
    b.Parent = Main
    local ui = Instance.new("UICorner")
    ui.CornerRadius = UDim.new(0, 8)
    ui.Parent = b
    b.MouseButton1Click:Connect(btn.Func)
end
